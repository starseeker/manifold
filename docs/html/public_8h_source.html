48inline float sind(float x) {
49 if (!std::isfinite(x)) return sin(x);
50 if (x < 0.0f) return -sind(-x);
51 int quo;
52 x = remquo(fabs(x), 90.0f, &quo);
53 switch (quo % 4) {
54 case 0:
55 return sin(glm::radians(x));
56 case 1:
57 return cos(glm::radians(x));
58 case 2:
59 return -sin(glm::radians(x));
60 case 3:
61 return -cos(glm::radians(x));
62 }
63 return 0.0f;
64}

79inline glm::mat4x3 RotateUp(glm::vec3 up) {
80 up = glm::normalize(up);
81 glm::vec3 axis = glm::cross(up, {0, 0, 1});
82 float angle = glm::asin(glm::length(axis));
83 if (glm::dot(up, {0, 0, 1}) < 0) angle = glm::pi<float>() - angle;
84 return glm::mat4x3(glm::rotate(glm::mat4(1), angle, axis));
85}

98inline int CCW(glm::vec2 p0, glm::vec2 p1, glm::vec2 p2, float tol) {
99 glm::vec2 v1 = p1 - p0;
100 glm::vec2 v2 = p2 - p0;
101 float area = v1.x * v2.y - v1.y * v2.x;
102 float base2 = glm::max(glm::dot(v1, v1), glm::dot(v2, v2));
103 if (area * area * 4 <= base2 * tol * tol)
104 return 0;
105 else
106 return area > 0 ? 1 : -1;
107}

501class Quality {
502 private:
503 inline static int circularSegments_ = 0;
504 inline static float circularAngle_ = 10.0f;
505 inline static float circularEdgeLength_ = 1.0f;
506
507 public:
518 static void SetMinCircularAngle(float angle) {
519 if (angle <= 0) return;
520 circularAngle_ = angle;
521 }
522
532 static void SetMinCircularEdgeLength(float length) {
533 if (length <= 0) return;
534 circularEdgeLength_ = length;
535 }
536
546 static void SetCircularSegments(int number) {
547 if (number < 3 && number != 0) return;
548 circularSegments_ = number;
549 }
550
558 static int GetCircularSegments(float radius) {
559 if (circularSegments_ > 0) return circularSegments_;
560 int nSegA = 360.0f / circularAngle_;
561 int nSegL = 2.0f * radius * glm::pi<float>() / circularEdgeLength_;
562 int nSeg = fmin(nSegA, nSegL) + 3;
563 nSeg -= nSeg % 4;
564 return std::max(nSeg, 3);
565 }
566};
