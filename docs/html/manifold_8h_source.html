49template <typename Precision, typename I = uint32_t>
50struct MeshGLP {
52 I NumVert() const { return vertProperties.size() / numProp; };
54 I NumTri() const { return triVerts.size() / 3; };
56 I numProp = 3;
60 std::vector<Precision> vertProperties;
63 std::vector<I> triVerts;
66 std::vector<I> mergeFromVert;
70 std::vector<I> mergeToVert;
78 std::vector<I> runIndex;
84 std::vector<uint32_t> runOriginalID;
89 std::vector<Precision> runTransform;
94 std::vector<I> faceID;
99 std::vector<Precision> halfedgeTangent;
105 Precision tolerance = 0;
106
107 MeshGLP() = default;
108
109 bool Merge();
110
111 la::vec<Precision, 3> GetVertPos(size_t i) const {
112 size_t offset = i * numProp;
113 return la::vec<Precision, 3>(vertProperties[offset],
114 vertProperties[offset + 1],
115 vertProperties[offset + 2]);
116 }
117
118 la::vec<I, 3> GetTriVerts(size_t i) const {
119 size_t offset = 3 * i;
120 return la::vec<I, 3>(triVerts[offset], triVerts[offset + 1],
121 triVerts[offset + 2]);
122 }
123
124 la::vec<Precision, 4> GetTangent(size_t i) const {
125 size_t offset = 4 * i;
126 return la::vec<Precision, 4>(
127 halfedgeTangent[offset], halfedgeTangent[offset + 1],
128 halfedgeTangent[offset + 2], halfedgeTangent[offset + 3]);
129 }
130};
